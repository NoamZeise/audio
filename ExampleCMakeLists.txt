cmake_minimum_required(VERSION 3.10)

set(ProjectName "Audio")
# set the project name
project(${ProjectName} VERSION 1.0)

file(GLOB SOURCES src/*)
add_executable(${ProjectName} ${SOURCES}
                              ${PROJ_INCLUDE})

if(UNIX)
    target_link_libraries(${ProjectName} sndfile)
    target_link_libraries(${ProjectName} portaudio)
elseif(MSVC)
  target_include_directories(${ProjectName} PUBLIC YOUR/PATH/TO/INCLUDE/DIRECTIORIES)
  target_link_libraries(${ProjectName} YOUR/PATH/TO/portaudio_x64.lib
                                       YOUR/PATH/TO/sndfile.lib)
endif()
